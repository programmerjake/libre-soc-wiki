\frame{\frametitle{Why create a Libre-Licensed ASIC?}

\vspace{15pt}

 \begin{itemize}
   \item Security, auditability, Sovereignty\vspace{6pt}
   \item Security through obscurity is always broken (eventually)
   		 \vspace{10pt}
   \item Fully Libre-Licensed including the VLSI toolchain:\\
   	 	 there's nowhere to hide vulnerabilitities\\
		\vspace{10pt}
   \item EU Digital Sovereignty: LIP6 Labs coriolis2\\
         is Europe's only Silicon-proven toolchain.\\
         OpenLANE: DARPA. Cadence VLSI etc. are US-owned.\\
		\vspace{10pt}
  \end{itemize}
}
